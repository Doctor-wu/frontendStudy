##   正则表达式之千分符 

```javascript
let num = "1829403203403";

(function(){
    function millimeter(){  //   ?= 正向预查
        return this.replace(/\d{1,3}(?=(\d{3})+$)/g,content=>content+",");
    }
    
    
    ["millimeter"].forEach(item=>{
        String.prototype[item] = eval(item);
    })
})()


num.millimeter(); // "1,829,403,203,403"
```
